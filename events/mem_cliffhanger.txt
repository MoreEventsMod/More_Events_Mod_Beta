fleet_event = {
	id = mem_cliffhanger.4
	title = mem_cliffhanger.4.name
	desc = mem_cliffhanger.4.desc

	archaeology = yes
	is_triggered_only = yes

	picture = GFX_evt_analyzing_artifacts
	show_sound = event_dig_site

	immediate = {
		from = { set_site_progress_locked = yes }
		IF = {
			limit = {
				owner = { is_gestalt = no }
			}
			leader = {
				add_trait = leader_trait_paranoid
			}
		}
	}

	after = { 
		from = { set_site_progress_locked = no }
		
		IF = {
			limit = {
				owner = { is_gestalt = no }
			}
			tooltip = {
				leader = {
					add_trait = leader_trait_paranoid
				}
			}
		}
		
		owner = {
			locked_random_list = {
				33 = {
					switch = {
						trigger = has_technology
						tech_housing_2 = {
							add_monthly_resource_mult = {
								resource = engineering_research
								value = @tier4researchreward
								min = @tier4researchmin
								max = @tier4researchmax
							}
						}
						tech_housing_1 = {
							add_research_option = tech_housing_2
							add_tech_progress = {
								tech = tech_housing_2
								progress = 0.5								
							}
						}
						default = {
							add_research_option = tech_housing_1
							add_tech_progress = {
								tech = tech_housing_1
								progress = 0.5								
							}
						}
					}
					modifier = {
						factor = 0
						OR = {
							is_gestalt = yes
							has_valid_civic = civic_agrarian_idyll
						}
					}
				}
				33 = {
					switch = {
						trigger = has_technology
						tech_ship_armor_4 = {
							add_monthly_resource_mult = {
								resource = engineering_research
								value = @tier4researchreward
								min = @tier4researchmin
								max = @tier4researchmax
							}
						}
						tech_ship_armor_3 = {
							add_research_option = tech_ship_armor_4
							add_tech_progress = {
								tech = tech_ship_armor_4
								progress = 0.5								
							}
						}
						tech_ship_armor_2 = {
							add_research_option = tech_ship_armor_3
							add_tech_progress = {
								tech = tech_ship_armor_3
								progress = 0.5								
							}
						}
						default = {
							add_research_option = tech_ship_armor_2
							add_tech_progress = {
								tech = tech_ship_armor_2
								progress = 0.5								
							}
						}
					}
				}
				33 = {
					IF = {
						limit = {
							has_technology = tech_crystal_armor_1
						}
						add_monthly_resource_mult = {
							resource = engineering_research
							value = @tier4researchreward
							min = @tier4researchmin
							max = @tier4researchmax
						}
					}
					ELSE = {
						add_research_option = tech_crystal_armor_1
						add_tech_progress = {
							tech = tech_crystal_armor_1
							progress = 0.5								
						}
					}
				}
				33 = {
					switch = {
						trigger = has_technology
						tech_mining_3 = {
							add_monthly_resource_mult = {
								resource = engineering_research
								value = @tier4researchreward
								min = @tier4researchmin
								max = @tier4researchmax
							}
						}
						tech_mining_2 = {
							add_research_option = tech_mining_3
							add_tech_progress = {
								tech = tech_mining_3
								progress = 0.5								
							}
						}
						tech_mining_1 = {
							add_research_option = tech_mining_2
							add_tech_progress = {
								tech = tech_mining_2
								progress = 0.5								
							}
						}
						default = {
							add_research_option = tech_mining_1
							add_tech_progress = {
								tech = tech_mining_1
								progress = 0.5								
							}
						}
					}
				}
			}
		}
	}
	
	option = {
		name = mem_cliffhanger.4.a
		trigger = {
			owner = {
				is_xenophobe = no
				is_xenophile = no
				NOR = {
					has_valid_civic = civic_machine_terminator
					has_valid_civic = civic_machine_servitor
					has_valid_civic = civic_hive_devouring_swarm
				}
			}
		}
	}

	option = {
		name = mem_cliffhanger.4.a.phobe
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_fanatic_xenophobe
					has_ethic = ethic_xenophobe
					has_valid_civic = civic_machine_terminator
				}
			}
		}
	}

	option = {
		name = mem_cliffhanger.4.a.phile
		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_fanatic_xenophile
					has_ethic = ethic_xenophile
					has_valid_civic = civic_machine_servitor
				}
			}
		}
	}

	option = {
		name = mem_cliffhander.4.a.swarm
		trigger = {
			owner = {
				has_valid_civic = civic_hive_devouring_swarm
			}
		}
	}
}
