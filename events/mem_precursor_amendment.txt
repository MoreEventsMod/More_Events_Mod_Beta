## This file contains events that interact with the vanilla precursor
## events to increase players' opportunities to complete them. A
## player who starts a precursor chain will generally be able to
## eventually complete it, although it may take a while, especially
## with a small empire.

## These events are numbered with the number of the anomaly event that
## they are mimicking. So for example mem_precursor_amendment.5
## is used to spawn precursor_events.5's project on an owned planet.

namespace = mem_precursor_amendment

### The Vultaum Star Assembly

# They have numbers 5; 10; 15; 20; 25; 30; 35; 40; 45; 50; 55

# Mining Base
planet_event = {
	id= mem_precursor_amendment.5
	title = mem_precursor_amendment.vultaum.title
	desc = mem_precursor_amendment.vultaum.desc
	picture = GFX_evt_archaeological_dig
	
	trigger = {
		is_capital = no
		exists = owner
		owner = {
			is_ai = no
			has_event_chain = vultaum_chain
			NOT = { 
				has_country_flag = vultaum_system 
				has_country_flag = vultaum_1
				has_country_flag = mem_recent_precursor_find
			} 
		}
	}
	
	mean_time_to_happen = {
		months = 3600 #at 35-44 years, with 20 planets, 5 unfound precursor events, average 1 per 12 years
		modifier = {
			factor = 10
			years_passed < 10
		}
		modifier = {
			factor = 8
			years_passed < 20
		}
		modifier = {
			factor = 6
			years_passed < 25
		}
		modifier = {
			factor = 3
			years_passed < 35
		}
		modifier = {
			factor = 2.5
			years_passed < 45
		}
		modifier = {
			factor = 1.6
			years_passed < 65
		}
		modifier = {
			factor = 0.8
			years_passed > 100
		}
		modifier = {
			factor = 0.6
			years_passed > 150
		}
	}
	
	immediate = {
		hidden_effect = {
			owner = { 
				set_country_flag = vultaum_1 
				set_timed_country_flag = { flag = mem_recent_precursor_find days = 1080 }
			}
		}
	}
	
	option = {
		name = FASCINATING
		enable_special_project = {
			name = "VULTAUM_ARCHEOLOGY_1"
			location = this
			owner = root.owner
		}
	}
}
