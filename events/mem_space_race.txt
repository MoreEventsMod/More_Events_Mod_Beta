#Gatekeeping event to control probabilities better without MTTH modifiers
country_event = {
	id = mem_space_race.1000
	hide_window = yes
	
	trigger = {
		NOT = { has_global_flag = mem_space_race_happened }
		years_passed > 30
		is_country_type = default
		NOR = {
			has_country_flag = mem_space_race_declined
			has_ethic = ethic_gestalt_consciousness
			has_ethic = ethic_fanatic_xenophobe
			has_civic = civic_fanatic_purifiers
		}
		is_at_war = no
		OR = {
			is_ai = no
			any_country = {
				is_ai = no
				is_at_war = no
				has_communications = root
			}
		}
		count_country = {
			count > 6
			limit = {
				is_country_type = default
				has_communications = root
				is_at_war = no
				NOR = {
					has_civic = civic_fanatic_purifiers
					has_civic = civic_machine_terminator
					has_civic = civic_hive_devouring_swarm
				}
			}
		}
	}
	
	mean_time_to_happen = { years = 50 }
	
	immediate = {
		random_list = {
			20 = {
				modifier = {
					factor = 9
					is_ai = yes
				}
			}
			10 = { 
				country_event = { id = mem_space_race.1 } 
				modifier = {
					factor = 0.5
					has_ethic = ethic_xenophobe
				}
				modifier = {
					factor = 2
					OR = {
						has_ethic = ethic_xenophile
						has_ethic = ethic_fanatic_xenophile
					}
				}
				modifier = {
					factor = 1.5
					OR = {
						has_ethic = ethic_materialist
						has_ethic = ethic_fanatic_materialist
						has_ethic = ethic_egalitarian
						has_ethic = ethic_fanatic_egalitarian
					}
					NOT = { has_ethic = ethic_xenophobe }
				}
			}
			10 = {
				set_global_flag = mem_space_race_happened #No event in this game
				modifier = {
					factor = 0.5
					has_ethic = ethic_xenophobe
				}
				modifier = {
					factor = 2
					OR = {
						has_ethic = ethic_xenophile
						has_ethic = ethic_fanatic_xenophile
					}
				}
				modifier = {
					factor = 1.5
					OR = {
						has_ethic = ethic_materialist
						has_ethic = ethic_fanatic_materialist
						has_ethic = ethic_egalitarian
						has_ethic = ethic_fanatic_egalitarian
					}
					NOT = { has_ethic = ethic_xenophobe }
				}
			}
		}
	}
}
