country_event = {
	id = socengdec.12
	title = "socengdec.12.name"
	desc = "socengdec.12.desc"
	picture = GFX_evt_crisis_defeated
	show_sound = event_celebration

	is_triggered_only = yes

	trigger = {
		mid_game_years_passed >= 50
		years_of_peace = { value > 50 }
		num_rivals < 1
		NOT = {
			any_neighbor_country = {
				NOT = { root = { is_overlord_to = prev } }
				is_country_type = default
				relative_power = {
					who = root
					value > inferior
				}
			}
		}
		NOR = {
			has_deficit = alloys
			has_deficit = consumer_goods
			has_deficit = energy
			has_deficit = minerals
		}
		NOR = {
			has_modifier = "soceng_imperial_decadence_x"
			has_modifier = "soceng_imperial_decadence_ix"
			has_modifier = "soceng_imperial_decadence_viii"

	desc = "socengdec.16.desc"

		is_homeworld = no
		has_ground_combat = no
		is_capital = no
		is_occupied_flag = no

		num_pops > 10
		has_modifier = "soceng_provincialism_x_mod"
		NOT = { has_modifier = "soceng_seperatism_mod" }
		OR = {
			AND = {
				NOT = { sector = { has_sector_type = core_sector } }
				owner = {
					has_policy_flag = soceng_interstellar_administration_flag
					num_sectors > 3
				}
			}
			owner = { has_policy_flag = soceng_planetary_administration_flag }
		}
		owner = {
			NOR = {
				has_modifier = "soceng_imperial_decadence_x"
				has_modifier = "soceng_imperial_decadence_ix"
				has_modifier = "soceng_imperial_decadence_viii"
				has_modifier = "soceng_imperial_decadence_vii"
				has_modifier = "soceng_imperial_decadence_vi"
			}
		}
	}

	is_triggered_only = yes

	immediate = {
		add_modifier = {
			modifier = "soceng_seperatism_mod"
			days = -1
		}
	}

	option = {
		name = socengdec.16.a
	}
}
