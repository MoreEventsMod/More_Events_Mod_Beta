fleet_event = {
    id = mem_through_the_fog.4
    title = mem_through_the_fog.4.name
    desc = mem_through_the_fog.4.desc

    archaeology = yes
    is_triggered_only = yes

    picture = GFX_evt_physics_research
    show_sound = event_radio_chatter

    immediate = {
		from = { set_site_progress_locked = yes }
	}
    
	after = { 
        from = { set_site_progress_locked = no }
        owner = {
            if = {
                limit = { has_technology = tech_engineering_3 }
                add_monthly_resource_mult = {
                    resource = engineering_research
                    value = @tier3researchreward
                    min = @tier3researchmin
                    max = @tier3researchmax
                }
            }
            else_if = {
                limit = { has_technology = tech_engineering_2 }
                add_research_option = tech_engineering_3
            }
            else_if = {
                limit = { has_technology = tech_engineering_1 }
                add_research_option = tech_engineering_2
            }
            else = {
                add_research_option = tech_engineering_1
            }
            if = {
                limit = {
                    has_distar = yes
                    NOT = { has_global_flag = l_cluster_opened }
                    has_event_chain = l_cluster_chain
                }
                generate_lcluster_clue = yes
            }
            else = {
                prev = {
                    large_artifact_reward = yes
                }
            }
        } 
    }

    option = {
        name = FASCINATING
        owner = {
            add_monthly_resource_mult = {
				resource = society_research
				value = @tier3researchreward
				min = @tier3researchmin
				max = @tier3researchmax
			}
        }

        trigger = {
            owner = {
                NOR = {
                    has_valid_civic = civic_fanatic_purifiers
                    has_valid_civic = civic_machine_terminator
                }
            }
        }
    }

    option = {
        name = mem_through_the_fog.4.a
        owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = @tier3influencereward
				min = @tier3influencemin
				max = @tier3influencemax
			}
        }
        
        trigger = {
            owner = {
                OR = {
                    has_valid_civic = civic_fanatic_purifiers
                    has_valid_civic = civic_machine_terminator
                    is_xenophobe = yes
                }
            }
        }
    }
}
