# Weapon Platform Deployment Rig
# By Violent Beetle

namespace = mem_wpdr
@max_support_ships = 15
# Rig's spawning loop
country_event = {
	id = mem_wpdr.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = {
			has_global_flag = mem_wpdr_ship_disabled
		}
	}

	immediate = {
		random_owned_fleet = {
			IF = {
				limit = {
					OR = {
						count_fleet_ships = {
							limit = {
								is_ship_size = mem_wpdr_tiny_ship
							}
							count < @max_support_ships
						}
						is_in_combat = yes
					}
				}
				mem_wpdr_create_random_platform = yes
			}
			ELSE_IF = {
				limit = {
					count_fleet_ships = {
						limit = {
							is_ship_size = mem_wpdr_tiny_ship
						}
						count > @max_support_ships
					}
				}
				random_owned_ship = {
					limit = {
						is_ship_size = mem_wpdr_tiny_ship
					}
					destroy_ship = this
				}
			}
		}
		country_event = {
			id = mem_wpdr.1
			days = 3
		}
	}
}

# Trigger for ship disabling
ship_event = {
	id = mem_wpdr.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_ship_size = mem_wpdr_capital_ship
	}

	immediate = {
		save_event_target_as = capital_ship
		set_global_flag = mem_wpdr_ship_disabled
		owner = {
			every_owned_ship = {
				limit = {
					is_ship_size = mem_wpdr_tiny_ship
				}
				destroy_ship = this
			}
		}
		from.owner = {
			country_event = {
				id = mem_wpdr.3
			}
		}
	}
}

# Entering the system
ship_event = {
	id = mem_wpdr.5
	title = mem_wpdr.5.name
	desc = mem_wpdr.5.desc

	is_triggered_only = yes

	location = FROM
	picture = GFX_evt_unidentified_ship
	show_sound = event_default

	trigger = {
		from = {
			has_star_flag = mem_wpdr_system
			any_ship_in_system = {
				is_disabled = no
				owner = {
					is_country_type = mem_wpdr_country
				}
			}
		}
		owner = {
			NOT = {
				has_country_flag = mem_wpdr_met_the_thing
			}
		}
	}

	immediate = {
		owner = {
			set_country_flag = mem_wpdr_met_the_thing
			random_country = {
				limit = {
					is_country_type = mem_wpdr_country
				}
				establish_communications_no_message = prev
			}
		}
	}

	after = {
		hidden_effect = {
			owner = {
				country_event = { id = story.8 days = 15 }
			}
		}
	}

	option = {
		name = mem_wpdr.5.a
	}
}
