ship_event = {
	id = mem_lotc.1
	title = "mem_lotc.1.name"
	desc = "mem_lotc.1.desc"
	picture = "GFX_evt_derelict_interior"
	is_triggered_only = yes
	
	immediate = {
    	owner = {
    		set_country_flag = mem_lotc_finder
			# leads to story event "Remnants" (first contact stuff)
			country_event = { id = story.5 days = 30 }
    	}
		
		from = {
			set_planet_flag = mem_lotc_start
			save_event_target_as = start
			solar_system = {
				set_star_flag = mem_lotc_start_system
			}
		}
    }

	option = {
		name = mem_lotc.1a
		
		begin_event_chain = {
			event_chain = "mem_lotc_first_chain"
			target = root.owner
		}
		
		hidden_effect = {
			set_spawn_system_batch = begin
			FROM = {
				#First project
				closest_system = {
					limit = {
						NOR = {
							has_star_flag = hostile_system
							is_pirate_system = yes
						}
						has_access_fleet = root.owner
						any_system_planet = {
							OR = {
								is_planet_class = pc_ocean
								is_planet_class = pc_continental
								is_planet_class = pc_gaia
								is_planet_class = pc_tropical
							}
							NOT = { exists = owner }
						}
					}
					min_steps = 2
					max_steps = 5
					random_system_planet = {
						limit = {
							OR = {
								is_planet_class = pc_ocean
								is_planet_class = pc_continental
								is_planet_class = pc_gaia
								is_planet_class = pc_tropical
							}
							NOT = { exists = owner }
						}
						save_event_target_as = mem_lotc_first_planet
					}
				}
				IF = {
					limit = {
						NOT = { exists = event_target:mem_lotc_first_planet }
					}
					closest_system = {
						limit = {
							NOT = { has_star_flag = mem_lotc_used_as_spawner }
						}
						min_steps = 1
						set_star_flag = mem_lotc_used_as_spawner
						spawn_system = {
							max_distance <= 20
							min_distance >= 5
							initializer = "mem_lotc_first_init"
						}
					}
				}
				#Second project
				closest_system = {
					limit = {
						NOT = { has_star_flag = mem_lotc_used_as_spawner }
					}
					min_steps = 1
					set_star_flag = mem_lotc_used_as_spawner
					spawn_system = {
						max_distance <= 20
						min_distance >= 5
						initializer = "mem_lotc_second_init"
					}
				}
				#Third project
				closest_system = {
					limit = {
						NOR = {
							has_star_flag = hostile_system
							is_pirate_system = yes
							has_star_flag = mem_lotc_second
							any_system_planet = { 
								OR = {
									is_same_value = event_target:mem_lotc_first_planet 
								}
							}
						}
						has_access_fleet = root.owner
						any_system_planet = {
							is_planet_class = pc_gas_giant
						}
					}
					min_steps = 2
					max_steps = 5
					random_system_planet = {
						limit = { is_planet_class = pc_gas_giant }
						save_event_target_as = mem_lotc_third_planet
					}
				}
				IF = {
					limit = {
						NOT = { exists = event_target:mem_lotc_third_planet }
					}
					closest_system = {
						limit = {
							NOT = { has_star_flag = mem_lotc_used_as_spawner }
						}
						min_steps = 1
						set_star_flag = mem_lotc_used_as_spawner
						spawn_system = {
							max_distance <= 20
							min_distance >= 5
							initializer = "mem_lotc_third_init"
						}
					}
				}
				#Fourth project
				closest_system = {
					limit = {
						NOR = {
							has_star_flag = hostile_system
							is_pirate_system = yes
							has_star_flag = mem_lotc_second
							any_system_planet = { 
								OR = {
									is_same_value = event_target:mem_lotc_first_planet 
									is_same_value = event_target:mem_lotc_third_planet
								}
							}
						}
						has_access_fleet = root.owner
						any_system_planet = {
							OR = {
								is_planet_class = pc_barren
								is_planet_class = pc_barren_cold
							}
						}
					}
					min_steps = 2
					max_steps = 5
					random_system_planet = {
						limit = {
							OR = {
								is_planet_class = pc_barren
								is_planet_class = pc_barren_cold
							}
						}
						save_event_target_as = mem_lotc_fourth_planet
					}
				}
				IF = {
					limit = {
						NOT = { exists = event_target:mem_lotc_fourth_planet }
					}
					closest_system = {
						limit = {
							NOT = { has_star_flag = mem_lotc_used_as_spawner }
						}
						min_steps = 1
						set_star_flag = mem_lotc_used_as_spawner
						spawn_system = {
							max_distance <= 20
							min_distance >= 5
							initializer = "mem_lotc_fourth_init"
						}
					}
				}
				set_spawn_system_batch = end
			}

			#Project issuing
			random_system = {
				limit = {
					has_star_flag = mem_lotc_second
				}
				save_event_target_as = second_system
			}
			create_point_of_interest = {
				id = mem_lotc_poi.2
				name = "mem_lotc_poi.2"
				desc = "mem_lotc_poi.2.desc"
				event_chain = mem_lotc_first_chain
				location = event_target:second_system
			}
			owner = {
				set_country_flag = mem_lotc_going_for_it
			}

			enable_special_project = {
				name = "MEM_LOTC_FIRST"
				location = event_target:mem_lotc_first_planet
				owner = root
			}
			enable_special_project = {
				name = "MEM_LOTC_THIRD"
				location = event_target:mem_lotc_third_planet
				owner = root
			}
			enable_special_project = {
				name = "MEM_LOTC_FOURTH"
				location = event_target:mem_lotc_fourth_planet
				owner = root
			}
		}
	}

	option = {
		name = NOTIME
		owner = { 
			add_resource = {
				influence = 250
			}
		}
	}
}
