	location = FROM
	
	immediate = {
		from = {
			clear_deposits = yes
		}
		owner = { country_event = { id = story.5 days = 30 } }
		fleet = {

		}
	}
	
	option = {
		name = mem_refuel.1.a
		
		add_modifier = {
			modifier = mem_refuel_modifier
			days = 1080
		}
		custom_tooltip = mem_refuel.1.a.tooltip
		hidden_effect = {
			random_list = {
				33 = { ship_event = { id = mem_refuel.2 days = 1 random = 1079 } }
				66 = {  }
			}
		}
	}
	
	option = {
		name = mem_refuel.1.b
		
		FROM = {
			add_deposit = d_energy_3
		}
	}
	
	option = {
		name = mem_refuel.1.c
		
		trigger = {
			exists = leader
			leader = { has_trait = leader_trait_expertise_voidcraft }
		}
		
		FROM = { 
			add_deposit = d_engineering_3
			add_deposit = d_physics_3
		}
		leader = { add_experience = 100 }
	}
	
	option = {
		name = mem_refuel.1.d
		
		trigger = {
			exists = leader
			NOT = { leader = { has_trait = leader_trait_expertise_voidcraft } }
		}
		
		leader = { 
			add_trait = leader_trait_expertise_voidcraft 
		}
	}
	
	option = {
		name = mem_refuel.1.e
		
		owner = { 
			add_monthly_resource_mult = {
				resource = influence
				value = @tier1influencereward
				min = @tier1influencemin
				max = @tier1influencemax
			}
		}
	}
	
	after = {
		hidden_effect = {
			fleet = {
				set_event_locked = no
			}
		}
	}
}
