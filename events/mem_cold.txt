fleet_event = {
	id = mem_cold.4
	title = mem_cold.4.name
	desc = mem_cold.4.desc

	is_triggered_only = yes
	archaeology = yes

	picture = GFX_evt_mem_rift
	show_sound = event_dig_site

	immediate = {
		from = { set_site_progress_locked = yes }
		leader = {
			IF = {
				limit = {
					species = { is_organic_species = yes }
				}
				add_trait = mem_cold_irradiated_trait
				add_ruler_trait = mem_cold_irradiated_trait
			}
			IF = {
				limit = {
					NOT = {
						has_trait = leader_trait_expertise_particles
					}
				}
				add_trait = leader_trait_expertise_particles
			}
			add_experience = 750
		}
	}

	after = { 
		from = { set_site_progress_locked = no }
		tooltip = {
			leader = {
				IF = {
					limit = {
						species = { is_organic_species = yes }
					}
					add_trait = mem_cold_irradiated_trait
				}
				IF = {
					limit = {
						NOT = {
							has_trait = leader_trait_expertise_particles
						}
					}
					add_trait = leader_trait_expertise_particles
				}
				add_experience = 750
			}
		}
		owner = {
			IF = {
				limit = {
					has_technology = null_void_beam
				}
				add_monthly_resource_mult = {
					resource = physics_research
					value = @tier5researchreward
					min = @tier5researchmin
					max = @tier5researchmax
				}
			}
			ELSE = {
				add_research_option = null_void_beam
				add_tech_progress = {
					tech = null_void_beam
					progress = 0.75
				}
			}
		}
	}

	option = {
		name = mem_cold.4.a
	}
}
