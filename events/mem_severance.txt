# The loop for the final encounter
planet_event = {
    id = mem_severance.7
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        random_list = {
            95 = {
                planet_event = { id = mem_severance.7 days = @loop_length }
                modifier = {
                    factor = 0.1
                    NOT = { has_planet_flag = mem_severance_time_to_sudden_death }
                }
            }
            5 = {
                planet_event = { id = mem_severance.20 }
                modifier = {
                    factor = 0
                    NOT = { exists = owner }
                }
                modifier = {
                    factor = 0
                    num_pops < 5
                }
                modifier = {
                    factor = 2
                    num_pops > 10
                }
                modifier = {
                    factor = 3
                    num_pops > 40
                }
                modifier = {
                    factor = 4
                    num_pops > 80
                }
            }
        }
    }
}
