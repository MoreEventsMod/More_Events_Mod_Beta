# AI Routines for Endless Sea

namespace = mem_colonizers_ai
@looplength = 10
# AI routine for mid game - main loop
country_event = {
	id = mem_colonizers_ai.1000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = mem_colonizers_dormant_country
		has_country_flag = mem_colonizers_begin_expansion
	}

	immediate = {
		IF = {
			limit = {
				NOT = {
					any_system = {
						has_star_flag = mem_colonizers_expansion_desire
					}
				}
			}
			random_system = {
				limit = {
					any_neighbor_system = {
						exists = space_owner
						owner = { is_same_value = root }
					}
				}
				set_star_flag = mem_colonizers_expansion_desire
			}
		}
		ELSE = {
			
		}
		country_event = {
			id = mem_colonizers_ai.1000
			days = @looplength
		}
	}
}
