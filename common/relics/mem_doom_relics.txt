@activation_cost = 150
@triumph_duration = 3600
r_mem_doom_capacitor = {
    activation_duration = @triumph_duration
    portrait = GFX_mem_doom_capacitor
    sound = event_activating_unknown_technology

    resources = {
		category = relics
		# Activation cost
		cost = {
			influence = @activation_cost
        }
        produces = {
            energy = 30
        }
    }

    ai_weight = {
        weight = 100
    }

    triggered_country_modifier = {
		potential = {
			always = yes
        }
    }
    
    score = 500 # Almost as good as great khan

    active_effect = {
        custom_tooltip = mem_doom_capacitor_effect_tooltip
        hidden_effect = {
            create_fleet = {
                effect = {
                    set_owner = prev
                    create_ship = {
                        name = "Maledict"
                        design = NAME_MEM_doom_dragon_design
                    }
                    set_location = prev.capital_scope
                }
            }
        }
		custom_tooltip = relic_triumph_cooldown
		hidden_effect = {
			add_modifier = {
				modifier = "relic_activation_cooldown" 
				days = @triumph_duration
			}		
		}
    }

    possible = {
		custom_tooltip = {
			fail_text = "requires_relic_no_cooldown"
			NOT = { has_modifier = relic_activation_cooldown }
        }
        custom_tooltip = {
            fail_text = {
                text = "mem_doom_requires_no_other_ship"
                NOT = {
                    any_owned_ship = {
                        is_ship_size = mem_doom_dragon_ship
                    }
                }
            }
        }
	}
}
