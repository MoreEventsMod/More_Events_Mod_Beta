#Ex.: "He is a student => "Er ist ein student[X] / Sie ist eine student[in]."

	name = GetXInDE
	text = {
		trigger = { cc_loc_leader_masc = yes }
		localization_key = DELOC_X
	}
	text = {
		trigger = { cc_loc_leader_fem = yes }
		localization_key = DELOC_in
	}
}
