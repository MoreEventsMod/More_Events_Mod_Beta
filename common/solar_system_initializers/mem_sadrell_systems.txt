    class = sc_binary_1
    usage = misc_system_init
    flags = { mem_sadrell_gate_system mem_sadrell_home_cluster }
    usage_odds = {
        base = 50000
        modifier = {
	        factor = 0
	        has_star_flag = empire_cluster
        }
        modifier = {
			factor = 0
			is_fe_cluster = yes
		}
    }
    max_instances = 1
    scaled_spawn_chance = 8
    asteroid_belt = {
        type = rocky_asteroid_belt
        radius = 100
    }

    change_orbit = 25

    planet = {
        class = star
        orbit_distance = 0
        orbit_angle = { min = 250 max = 290 }
        size = 25
    }

    planet = {
        class = star
        orbit_distance = 0
        orbit_angle = 180
        size = 16
    }

    planet = {
        class = pc_broken
        orbit_distance = 50
    }

    change_orbit = 25

    planet = {
        count = 4
        class = pc_asteroid
        orbit_distance = 0
    }

    change_orbit = 10

    planet = {
        count = 3
        class = random_non_colonizable
        orbit_distance = 25

        moon = {
            count = { min = 0 max = 1 }
            class = random_non_colonizable
            orbit_distance = 10
        }
    }

    init_effect = {
        set_global_flag = mem_sadrell_cluster_spawned
        every_neighbor_system = {
            limit = {
                NOR = {
                    has_star_flag = empire_cluster
                    is_fe_cluster = yes
                    has_star_flag = hostile_system
                    has_star_flag = marauder_system
                }
            }
            set_star_flag = mem_sadrell_home_cluster
			every_neighbor_system = {
                limit = {
                    NOR = {
                        has_star_flag = empire_cluster
                        is_fe_cluster = yes
                        has_star_flag = hostile_system
                        has_star_flag = marauder_system
                    }
                }
                set_star_flag = mem_sadrell_home_cluster
			}
        }
        every_system = { # This should discourage leviathans and the like from spawning in this cluster
            limit = { has_star_flag = mem_sadrell_home_cluster }
            set_star_flag = empire_cluster
        }
        spawn_megastructure = {
			type = mem_sadrell_gateway_inactive
			orbit_angle = 0
			orbit_distance = 60
        }
    }
}
