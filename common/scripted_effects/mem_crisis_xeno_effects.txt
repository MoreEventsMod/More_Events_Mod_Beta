### Xeno crisis scripted effects

@days = 500
@randomDays = 250

# This checks to determine whether or not the crisis should continue when a conservative option is chosen. Can't make progress without a bit of risk...
mem_crisis_xeno_keep_going = {
    hidden_effect = {
        random_list = {
            # Unfortunately for the player, the crisis continues.
            20 = {
                owner = {
                    country_event = { id = $EVENT$ days = @days random = @randomDays }
                }
            }
            # Atomizing the specimen proved fruitful in stopping the crisis.
            80 = {}
        }
    }
}
