# Anomaly Category

anomaly_category = {
	key = mem_nanobot_room_category
	desc = mem_nanobot_room_category_desc
	picture = GFX_evt_ship_in_orbit
	level = 2
	
	spawn_chance = {
		modifier = {
			OR = {
				is_asteroid = yes
				is_planet_class = pc_barren
				is_planet_class = pc_barren_cold
				is_planet_class = pc_frozen
			}
			add = 2 
		}
		modifier = {
			factor = 0
			from.owner = { has_country_flag = mem_nanobot_room_spawned }
		}
	}
	
	on_spawn = {
		from.owner = { set_country_flag = mem_nanobot_room_spawned }
	}
	
	on_success = {
	
	}
	
	on_fail = {
		ship_event = {
			id = mem_nanobot_room.10
		}
	}
	
	on_critical_fail = {
		ship_event = {
			id = mem_nanobot_room.20
		}
	}
}
