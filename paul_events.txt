country_event = {
	id = paul.1
	title = "paul.1.name"
	desc = "paul.1.desc"
	picture = GFX_evt_star_chart
	show_sound = event_laboratory_sound
	trigger = {
		is_ai = no
		NOT = { has_country_flag = paul_star_survey }	
	}

		months = 60
	}

		set_country_flag = paul_star_survey
	}

	option = {
		name = "paul.1.a"
		begin_event_chain = {
			event_chain = "paul_star_chain"
			target = root
		}
		hidden_effect = {
			closest_system = {
				limit = {
					is_star_class = sc_a
				}
				random_system_planet = {
					limit = {
						is_planet_class = pc_a_star
					} 
					enable_special_project = {
						name = "A_STAR_SURVEY"
						location = this
						owner = root
					}		 
				}
				
			}
			closest_system = {
				limit = {
					is_star_class = sc_k
				}
				random_system_planet = {
					limit = {
						is_planet_class = pc_k_star
					} 
					enable_special_project = {
						name = "K_STAR_SURVEY"
						location = this
						owner = root
					}		 
				}
				
			}
			closest_system = {
				limit = {
					is_star_class = sc_m
				}
				random_system_planet = {
					limit = {
						is_planet_class = pc_m_star
					} 
					enable_special_project = {
						name = "M_STAR_SURVEY"
						location = this
						owner = root
					}		 
				}
				
			}
			closest_system = {
				limit = {
					OR = {
						is_star_class = sc_black_hole
						is_star_class = sc_pulsar
						is_star_class = sc_neutron_star

				}

					limit = {
						OR = {
							is_planet_class = pc_neutron_star
							is_planet_class = pc_black_hole
							is_planet_class = pc_pulsar
						}
						
					} 
					enable_special_project = {
						name = "RARE_STAR_SURVEY"
						location = this
						owner = root
					}		 
				}
				
			}
		}
	}
	option = {
		name = "paul.1.b"
	}

						limit = {
							has_tech_option = tech_fusion_power
						}

							tech = tech_fusion_power

						}
						else = {
							add_research_option = tech_fusion_power
							add_tech_progress = {
								tech = tech_fusion_power
								progress = 0.25
							}
						}
					}

		}
	}
